48  template <class MAT>
49  auto operator()(MAT&& src, cv::Rect roi)
50  {
51  cv::Mat_<float> range = src(roi);
52  auto lower = percentiles(range, m.pctl[0]);
53  auto upper = percentiles(range, m.pctl[1]);
54  auto pixels = 0;
55  auto prev = 0.0;
56  auto mean = 0.0;
57  auto stddev = 0.0;
58  for (auto i = 0; i != range.total(); ++i)
59  {
60  auto value = range.template at<float>(i);
61  if (value >= lower and value <= upper)
62  {
63  prev = mean;
64  mean += (value - mean) / (++pixels);
65  stddev += (value - mean) * (value - prev);
66  }
67  }
68  stddev = std::sqrt(stddev / (pixels - 1));
69 
70  return ValueType(mean, stddev, pixels);
71  }

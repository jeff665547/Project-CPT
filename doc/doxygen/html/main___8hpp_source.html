501  private:
502  template <int32_t dim, int32_t lg2nfft>
503  static auto fit_sinewave(const cv::Mat_<FLOAT>& src)
504  {
505  // 1D projection
506  cv::Mat_<FLOAT> data;
507  cv::reduce(src, data, dim, CV_REDUCE_AVG, data.depth());
508  if (dim == 1)
509  cv::transpose(data, data);
510 
511  // Moving average
512  cv::Mat_<FLOAT> ac = cv::Mat_<FLOAT>::zeros(1, 1 << lg2nfft);
513  for (auto i = 2; i != data.total(); ++i)
514  ac(i - 2) = ( 2 * data(i) - data(i - 1) - data(i - 2) ) / 3;
515 
516  // Frequency detection
517  cv::Mat_<cv::Point_<FLOAT>> ft;
518  cv::dft(ac, ft, cv::DFT_COMPLEX_OUTPUT);
519  FLOAT max = 0.0;
520  FLOAT loc = 0.0;
521  auto last = ft.total() >> 1;
522  for (auto i = 0; i != last; ++i)
523  {
524  auto val = cv::norm(ft(i));
525  if (max < val)
526  {
527  max = val;
528  loc = i;
529  }
530  }
531  const double freq = loc / static_cast<FLOAT>(ft.total());
532  std::cerr << "intvl = " << 1.0 / freq << '\n';
533 
534  // Phase estimation
535  cv::Mat_<FLOAT> Phi, w;
536  const auto v = 2.0 * CV_PI * freq;
537  for (auto i = 0; i != data.total(); ++i)
538  {
539  cv::Mat_<FLOAT> phi = (
540  decltype(phi)(1, 3)
541  << std::cos(v * i)
542  , std::sin(v * i)
543  , 1.0
544  );
545  Phi.push_back(phi);
546  }
547  cv::transpose(data, data);
548  cv::solve(Phi, data, w, cv::DECOMP_NORMAL);
549  const double phase = std::atan2(w(0), w(1)) * 0.5 / CV_PI + 0.25;
550  std::cerr << "phase = " << phase << '\n';
551 
552  // Generate gridlines
553  std::vector<FLOAT> anchors;
554  int32_t start = std::ceil(phase);
555  int32_t end = std::floor(freq * data.total() + phase);
556  while (std::round((start - 1 - phase) / freq) > 0) --start;
557  while (std::round((end - phase) / freq) < data.total()) ++end;
558  for (auto i = start; i != end; ++i)
559  anchors.emplace_back((i - phase) / freq);
560 
561  return anchors;
562  }

608 class TrimmedMean
609 {
610  public:
611  using ValueType = std::tuple<float, float, uint16_t>;
612 
613  private:
614  double lowest_, highest_;
615 
616  public:
617  template <class DB>
618  TrimmedMean(DB& db)
619  : lowest_(db.lowest)
620  , highest_(db.highest)
621  {}
622 
623  auto operator()(cv::Mat& image, cv::Rect roi)
624  {
625  auto view = image(roi);
626 
627  std::sort(view.begin<float>(), view.end<float>());
628  auto l = static_cast<int>(lowest_ * view.total());
629  auto h = static_cast<int>(highest_ * view.total());
630 
631  float mean = std::accumulate(
632  view.begin<float>() + l
633  , view.begin<float>() + h
634  , 0.0
635  );
636  mean /= (h - l);
637 
638  float stddev = std::accumulate(
639  view.begin<float>() + l
640  , view.begin<float>() + h
641  , 0.0
642  , [mean](const auto& init, const auto& elem)
643  { return init + (elem - mean) * (elem - mean); }
644  );
645  stddev = std::sqrt(stddev / (h - l - 1));
646 
647 
648  return ValueType(mean, stddev, h - l);
649  }
